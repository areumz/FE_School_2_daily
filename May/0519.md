```js
function outer() {
    
    let privateVal = ['test'];
    
    function getPrivate() {
        
        return privateVal;

    }

    return getPrivate;

}

const private = outer();
let secretVal = private();

console.log(secretVal); //['test']
// ->이것으로 인해 outer 종료되었지만 메모리 상에서 사라지지 않고 있음
```
